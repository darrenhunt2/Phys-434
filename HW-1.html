# generate 100000 points for a normal distribution using scipy
points = stats.norm.rvs(loc = 5., scale = 0.01, size = 100000)
In [5]:
# plot a histogram of the points using matplotlib
fig, ax = plt.subplots(1, 1)  # one subplot (coordinates 1,1) -- change subplots for more plots
ax.hist(points,50, density=True)  # plot a histogram using the 100,000 points we generated above
# 50 bins, or classes, where the points are put into. these are equal widths
# density=True returns a probability density
plt.tick_params(labelsize = 24)  # changes the font size of the label
plt.xlim([4.95,5.05])  # changes the limits of the x axis on either side that display on the subplot
x = np.linspace(4.95,5.05,1000)  # returns 1000 equally space points between 4.95 and 5.05
ax.plot(x,stats.norm.pdf(x,loc = 5., scale = 0.01),linewidth = 8,alpha = 0.7)  # draws a normally
# distributed line over the same subplot using equally spaced points
plt.show()  # makes plot output visible
